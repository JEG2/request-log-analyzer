108:       # determine maximum cell widths
109:       max_cell_widths = rows.inject(Array.new(columns.length, 0)) do |result, row|
110:         lengths = row.map { |column| column.to_s.length }
111:         result.each_with_index { |length, index| result[index] = ([length, lengths[index]].max rescue length) }
112:       end
113:       columns.each_with_index { |col, index| col[:actual_width] ||= max_cell_widths[index] }
114:     
115:       # determine actual column width
116:       column_widths = columns.map do |column|
117:         if column[:width] == :rest
118:           nil
119:         elsif column[:width]
120:           column[:width]
121:         elsif column[:min_width]
122:           [column[:min_width], column[:actual_width]].max
123:         elsif column[:max_width]
124:           [column[:max_width], column[:actual_width]].min
125:         else
126:           column[:actual_width]
127:         end
128:       end
129:      
130:       if column_widths.include?(nil)
131:         fill_column = columns[column_widths.index(nil)]
132:         width_left = options[:width] - ((columns.length - 1) * (style[:cell_separator] ? 3 : 1)) - column_widths.compact.inject(0) { |sum, col| sum + col}
133:         column_widths[column_widths.index(nil)] = case fill_column[:type]
134:           when :ratio; width_left # max out
135:           else;        [width_left, fill_column[:actual_width]].min
136:         end
137:       end
138:     
139:       # Print table header
140:       if table_has_header?(columns)
141:         column_titles = []
142:         columns.each_with_index do |column, index|
143:           width = column_widths[index]        
144:           alignment = (column[:align] == :right ? '' : '-')
145:           column_titles.push(colorize("%#{alignment}#{width}s" % column[:title].to_s[0...width], :bold))
146:         end
147:       
148:         puts column_titles.join(style[:cell_separator] ? " #{characters[:vertical_line]} " : ' ')
149:         line(:green)
150:       end
151:     
152:       rows.each do |row|
153:         row_values = []
154:         columns.each_with_index do |column, index|
155:           width = column_widths[index]
156:           case column[:type]
157:           when :ratio
158:             if width > 4
159:               if column[:treshold] && column[:treshold] < row[index].to_f
160:                 bar = ''
161:                 bar << characters[:block] * (width.to_f * column[:treshold]).round 
162:                 bar << colorize(characters[:block] * (width.to_f * (row[index].to_f - column[:treshold])).round, :red) 
163:                 row_values.push(bar) 
164:               else
165:                 # Create a bar by combining block characters
166:                 row_values.push(characters[:block] * (width.to_f * row[index].to_f).round) 
167:               end
168:             else
169:               # Too few characters for a ratio bar. Display nothing
170:               row_values.push('')
171:             end
172:           else
173:             alignment = (columns[index][:align] == :right ? '' : '-')
174:             cell_value = "%#{alignment}#{width}s" % row[index].to_s[0...width]
175:             cell_value = colorize(cell_value, :bold, :brown) if columns[index][:highlight]
176:             row_values.push(cell_value)
177:           end
178:         end
179:         puts row_values.join(style[:cell_separator] ? " #{characters[:vertical_line]} " : ' ')      
180:       end
181:     end
